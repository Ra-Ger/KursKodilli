SELECT ISSUES.SUMMARY, ISSUES.DESCRIPTION, ISSUESLISTS.NAME
FROM ISSUES
JOIN ISSUESLISTS ON ISSUES.ISSUESLIST_ID = ISSUESLISTS.ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY, I.DESCRIPTION
FROM USERS U
JOIN ISSUES I ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASK_NUMBER
FROM USERS U JOIN ISSUES P ON U.ID = P.USER_ID_ASSIGNEDTO
GROUP BY U.ID, U.FIRSTNAME, U.LASTNAME;
